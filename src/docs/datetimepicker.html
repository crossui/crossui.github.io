datetimepicker 日期控件
开始时间到结束时间
HTML Javascript 

    <span>从</span>
    <input type="text" placeholder="" id="startDatePickerDemo" class="c-input c-input-medium date-picker">
    <span>到</span>
    <input type="text" placeholder="" id="endDatePickerDemo" class="c-input c-input-medium date-picker">



    $('#startDatePickerDemo').datetimepicker({
        format:'Y-m-d',
        onShow:function( ct ){
            this.setOptions({
                maxDate:$('#endDatePickerDemo').val()?$('#endDatePickerDemo').val():false
            })
        },
        timepicker:false
    });

    $('#endDatePickerDemo').datetimepicker({
        format:'Y-m-d',
        onShow:function( ct ){
            this.setOptions({
                minDate:$('#startDatePickerDemo').val()?$('#startDatePickerDemo').val():false
            })
        },
        timepicker:false
    });
