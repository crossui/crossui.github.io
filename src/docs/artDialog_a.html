关闭并返回数据到主页面
脚本 

    // 获取对话框传递过来的数据
    var dialog = top.dialog.get(window);
    var data = dialog.data;
    if (data) {
        $('#aInput').val(data);
    }

    // 关闭并返回数据到主页面
    $('#exit').on('click',function () {
        dialog.close($('#aInput').val()); // 关闭（隐藏）对话框
        dialog.remove();				 // 主动销毁对话框
    });
