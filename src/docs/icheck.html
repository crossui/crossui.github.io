    $('input.b').each(function(){
        var self = $(this),
        label = self.next(),
        label_text = label.text();

        label.remove();
        self.iCheck({
            checkboxClass: 'c-checkbox-line',
            radioClass: 'c-radio-line',
            insert: '<div class="line-icon"></div>' + label_text
        });
    });

方法

用法 描述
$('input').iCheck('check'); 将输入框的状态设置为checked
$('input').iCheck('uncheck'); 移除 checked 状态
$('input').iCheck('toggle'); 切换 checked 状态
$('input').iCheck('disable'); 将输入框的状态设置为 disabled
$('input').iCheck('enable'); 移除 disabled 状态
$('input').iCheck('indeterminate'); 改变 indeterminate 状态
$('input').iCheck('determinate'); 删除 indeterminate 状态
$('input').iCheck('update'); 应用输入更改，在插件外完成
$('input').iCheck('destroy'); 移除iCheck样式

回调事件

事件名称 使用时机
ifClicked 用户点击了自定义的输入框或与其相关联的label
ifChanged 输入框的 checked 或 disabled 状态改变了
ifChecked 输入框的状态变为 checked
ifUnchecked checked 状态被移除
ifDisabled 输入框状态变为 disabled
ifEnabled disabled 状态被移除
ifCreated 输入框被应用了iCheck样式
ifDestroyed iCheck样式被移除
使用on()方法绑定事件： 

    $('input').on('ifChecked', function(event){
        alert(event.type + ' callback');
    });
